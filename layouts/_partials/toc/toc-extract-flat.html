{{/* 从 .Content 提取扁平 header 列表 入参: .Content (页面 HTML 内容) 返回: slice of dict{ id, title, depth } */}} {{- $headers := findRE "]*>(.|\n)*?" . -}} {{- $items := slice -}} {{- range $headers -}} {{- $depth := substr . 2 1 | int -}} {{- $id := replaceRE `]*id="([^"]*)"[^>]*>(.|\n)*?` "$1" . -}} {{- $title := replaceRE `(.*?)` "$1" . | plainify -}} {{- $items = $items | append (dict "id" $id "title" $title "depth" $depth) -}} {{- end -}} {{- return $items -}}
