{{/* 生成嵌套 TOC 用法: {{ partial "toc.html" . }} */}} {{- $flat := partial "toc/toc-extract-flat.html" .Content -}} {{- if ge (len $flat) 2 -}} {{- $minDepth := 9 -}} {{- range $flat -}} {{- if lt .depth $minDepth -}}{{- $minDepth = .depth -}}{{- end -}} {{- end -}} {{- $tree := partial "toc/toc-build-tree" (dict "items" $flat "index" 0 "minDepth" $minDepth ) -}} {{- range $index, $node := $tree.nodes -}} {{- partial "toc/toc-heading-item.html" (dict "node" $node "index" $index) -}} {{- end -}}
{{- end -}}
